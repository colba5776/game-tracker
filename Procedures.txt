CREATE DEFINER=`root`@`localhost` PROCEDURE `ChangePTName`(IN inPTName varchar(45), IN inPTID int)
BEGIN
	update playthrough
    set playthroughName = inPTName
    where playthroughID = inPTID;
END


CREATE DEFINER=`root`@`localhost` PROCEDURE `ChangePTDesc`(IN inPTDesc varchar(45), IN inPTID int)
BEGIN
	update playthrough
    set playthroughDescription = inPTdesc
    where playthroughID = inPTID;
END


CREATE DEFINER=`root`@`localhost` PROCEDURE `changePTPERC`(IN inPTPERC INT, IN inPTID int)
BEGIN
	update playthrough
    set playthroughCurrentPercent = inPTPERC
    where playthroughID = inPTID;
END


CREATE DEFINER=`root`@`localhost` PROCEDURE `changePTPercTarg`(IN inPTPERCTarg INT, IN inPTID int)
BEGIN
	update playthrough
    set playthroughtargetpercent = inPTPERCTarg
    where playthroughID = inPTID;
END
